- 含纯虚函数的类是抽象类，不能实例化

- 泛化：将相同的上提、不同的下放
    Evt
    ├─ Evt1j
    │   └─ Evt1jDur
    │        └─ Rdv
    └─ EvtPj

    1. 提取共同属性
    2. 提取共同方法
    3. 扩展模型能力

    La généralisation consiste à abstraire les caractéristiques communes de plusieurs classes afin de définir une super-classe plus générale, permettant une modélisation plus cohérente et extensible du domaine.

@startuml
skinparam classAttributeIconSize 0

namespace TIME {

abstract class Evt {
  - sujet : string
  + Evt(s : string)
  + getDescription() : string
  {abstract} + afficher(f : ostream) : void
  + ~Evt()
}

class Evt1j {
  - date : Date
  + Evt1j(d : Date, s : string)
  + getDate() : Date
  + afficher(f : ostream) : void
}

class Evt1jDur {
  - debut : Horaire
  - duree : Duree
  + Evt1jDur(d:Date, s:string, h:Horaire, du:Duree)
  + getHoraire() : Horaire
  + getDuree() : Duree
  + afficher(f : ostream) : void
}

class Rdv {
  - personnes : string
  - lieu : string
  + Rdv(d:Date, s:string, h:Horaire, du:Duree, pers:string, l:string)
  + getPersonnes() : string
  + getLieu() : string
  + afficher(f : ostream) : void
}

class EvtPj {
  - debut : Date
  - fin : Date
  + EvtPj(d1:Date, d2:Date, s:string)
  + getDebut() : Date
  + getFin() : Date
  + afficher(f : ostream) : void
}

class Agenda {
  - evenements : vector<Evt*>
  + operator<<(e : Evt&) : Agenda&
  + afficher(f : ostream) : void
}

Evt <|-- Evt1j
Evt1j <|-- Evt1jDur
Evt1jDur <|-- Rdv
Evt <|-- EvtPj

Agenda o-- "0..*" Evt : agrégation

}
@enduml
